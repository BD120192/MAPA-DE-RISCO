CREATE TABLE SEXO (
    id serial PRIMARY KEY,
    genero varchar
);

CREATE TABLE VITIMA (
    id serial PRIMARY KEY,
    idade int,
    id_sexo serial,
    FK_SEXO_id serial,
    FK_CRIME_id serial
);

CREATE TABLE CRIME (
    data date,
    hora time,
    id_vitima serial,
    id serial PRIMARY KEY,
    id_rua serial,
    id_cidade serial,
    id_tipo_crime serial,
    id_bairro serial,
    FK_TIPO_DE_CRIME_id serial,
    FK_RUA_id serial
);

CREATE TABLE TIPO_DE_CRIME (
    descricao_crime varchar[100],
    id serial PRIMARY KEY
);

CREATE TABLE CIDADE (
    id serial PRIMARY KEY,
    nome_cidade varchar[100]
);

CREATE TABLE BAIRRO (
    id serial PRIMARY KEY,
    nome_bairro varchar[100],
    id_ciadade serial,
    FK_CIDADE_id serial
);

CREATE TABLE RUA (
    id_bairro serial,
    id_classificacao serial,
    nome_rua varchar[100],
    id serial PRIMARY KEY,
    quantidade_crimes integer,
    FK_BAIRRO_id serial,
    FK_CLASSIFICACAO_DESCRICAO_id serial
);

CREATE TABLE CLASSIFICACAO_DESCRICAO (
    tipo char,
    id_descricao serial,
    id serial PRIMARY KEY,
    textopadrao varchar[100]
);

CREATE TABLE LINHA (
    id serial PRIMARY KEY,
    numero_linha varchar[50],
    id_classificacao integer,
    FK_CLASSIFICACAO_DESCRICAO_id serial
);

CREATE TABLE USUARIO (
    nome char,
    senha char,
    cpf integer,
    email char,
    login char,
    id serial PRIMARY KEY
);

CREATE TABLE COMENTARIO (
    id serial PRIMARY KEY,
    comentario_usuario varchar[100],
    id_usuario serial,
    id_tipo_comentario serial,
    FK_TIPO_COMENTARIO_id serial,
    FK_RUA_id serial,
    FK_USUARIO_id serial,
    FK_LINHA_id serial
);

CREATE TABLE TIPO_COMENTARIO (
    id serial PRIMARY KEY,
    descricao_tipo_comentario varchar[100]
);

CREATE TABLE Linha_Rua (
    fk_LINHA_id serial,
    fk_RUA_id serial,
    id serial PRIMARY KEY
);
